    socket.on('mensajes',(mensajes)=>{

    })

    function render(mensajes){
    const html = mensajes.map((mensaje)=>{return `
        <div>
            <b>${mensaje.autor}:</b>
            <i>${mensaje.texto}</i>
        </div>`
    }).join(' ')

    document.getElementById('mensajes').innerHTML = html
}
